FORM CREATE_OBJECT .

* ## #### ##
  CREATE OBJECT GO_CUSTOM_CONTAINER
    EXPORTING
      CONTAINER_NAME = 'GO_CUSTOM_CONTAINER'.
* ## #### ## (MAIN CONTAINER ##)
  CREATE OBJECT GO_SPLITTER
    EXPORTING
      PARENT  = GO_CUSTOM_CONTAINER
      ROWS    = 1
      COLUMNS = 2.

* ## ### GUI CONTAINER ##.
  CALL METHOD GO_SPLITTER->GET_CONTAINER
    EXPORTING
      ROW       = 1
      COLUMN    = 1
    RECEIVING
      CONTAINER = GO_CONT1.

  CALL METHOD GO_SPLITTER->GET_CONTAINER
    EXPORTING
      ROW       = 1
      COLUMN    = 2
    RECEIVING
      CONTAINER = GO_CONT2.

* ### ### ##### GRID ##.
  CREATE OBJECT GO_GRID1
    EXPORTING
      I_PARENT = GO_CONT1.

  CREATE OBJECT GO_GRID2
    EXPORTING
      I_PARENT = GO_CONT2.


CALL METHOD GO_SPLITTER->SET_COLUMN_WIDTH
  EXPORTING
    ID          = 1
    WIDTH      = 36
.

ENDFORM.
